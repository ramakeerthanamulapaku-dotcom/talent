var page = 1; var perPage = 10; var data = books; var ascending = true; function displayTable() { let table = document.getElementById("books"); let rows = table.querySelectorAll("tr:not(:first-child)"); rows.forEach(r => r.remove()); let start = (page - 1) * perPage; let end = start + perPage; let pageData = data.slice(start, end); pageData.forEach(function(book) { let row = document.createElement("tr"); function cell(text) { let td = document.createElement("td"); td.innerText = text; return td; } row.appendChild(cell(book.title)); row.appendChild(cell(book.isbn)); row.appendChild(cell(book.pageCount)); row.appendChild(cell(book.thumbnailUrl)); row.appendChild(cell(book.status)); row.appendChild(cell(book.authors)); row.appendChild(cell(book.categories)); row.appendChild(cell(book.price)); row.appendChild(cell(book.currency)); row.appendChild(cell(book.discount)); row.appendChild(cell(book.discountUnits)); table.appendChild(row); }); document.getElementById("page").innerText = "Page " + page; } function nextPage() { if (page * perPage < data.length) { page++; showTable(); } } function previousPage() { if (page > 1) { page--; displayTable(); } } function sortTitle() { data.sort(function(a, b) { if (ascending) { return a.title > b.title ? 1 : -1; } else { return a.title < b.title ? 1 : -1; } }); ascending = !ascending; page = 1; displayTable(); } displayTable();
